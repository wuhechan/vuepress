# 设计模式分类

# 单例模式

text Ensure a class has only one instance，and provide a global point of access to it.(确保某一个类只有一个实例，而且自行实例化并向整个系统提供这个实例。)

# 单例模式代码例子

class Person {
    constructor(name, gender) {
        this.name = name
        this.gender = gender
    }
    know() {
        return '他是' + this.name +  '性别' + this.gender ;
    }
}
// 代理实现单例模式
var proxyMode = (function() {
    var instance = null;
    return function(name , gender) {
        if(!instance) {
            instance = new Person(name, gender);
        }
        // 这里牵扯到的知识点有闭包，由于 function 被外部环境引用instance始终存在于内存当中
        // 判断instance不等于null的时候直接去内存的实例，这样就达到了单例的效果
        return instance;
    }
})();

var a = proxyMode('小明', '男');
var b = proxyMode('小红', '女');
// 因为单例模式是只实例化一次，所以下面的实例是相等的
console.log(a === b); // true
console.log(a.name, b.name); // 小明, 小明
console.log(a.know(),b.know()) // 他是小明 性别是男性
